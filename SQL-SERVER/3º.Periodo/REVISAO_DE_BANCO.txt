CREATE TABLE CATEGORIA
(
  ID_categoria INT NOT NULL AUTO_INCREMENT,
  TIPO VARCHAR (45),
  CONSTRAINT PK_CATEGORIA PRIMARY KEY (ID_categoria)
);

CREATE TABLE MARCA
(
  ID_marca INT NOT NULL AUTO_INCREMENT,
  NOME VARCHAR (20),
  CONSTRAINT PK_MARCA PRIMARY KEY (ID_marca),
); 

CREATE TABLE PRODUTO
(
  ID_produto INT NOT NULL IDENTITY(1,1),-
  NOME VARCHAR (45) NOT NULL,
  PRECO DECIMAL(9,2),
  COD_CATEGORIA INT,
  COD_MARCA INT,
  CONSTRAINT PK_PRODUTO PRIMARY KEY (ID_produto),
  CONSTRAINT FK_PRODUTO_CATEGORIA FOREIGN KEY (COD_CATEGORIA) REFERENCES CATEGORIA (ID_categoria),
  CONSTRAINT FK_PRODUTO_MARCA FOREIGN KEY (COD_MARCA) REFERENCES MARCA (ID_marca)
);
