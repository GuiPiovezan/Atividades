CREATE TABLE SALAS (
ID_TURMA INT NOT NULL IDENTITY(1,1),
NOME VARCHAR(45),
NUMERO INT NOT NULL,
CONSTRAINT PK_SALA PRIMARY KEY(ID_SALA));

CREATE TABLE TURMAS (
ID_TURMA INT NOT NULL IDENTITY(1,1),
NOME VARCHAR(45) NOT NULL,
CONSTRAINT PK_TURMA PRIMARY KEY(ID_TURMA),
CONSTRAINT FK_SALA FOREIGN KEY (ID_SALA) REFERENCES (SALAS));

CREATE TABLE PROFESSORES (
ID_PROFESSOR INT NOT NULL IDENTITY(1,1),
NOME VARCHAR(45) NOT NULL,
CONSTRAINT PK_PROFESSOR PRIMARY KEY(ID_PROFESSOR));

CREATE TABLE ALUNOS (
ID_ALUNO INT NOT NULL IDENTITY(1,1),
NOME VARCHAR(45) NOT NULL,
CONSTRAINT PK_ALUNO PRIMARY KEY(ID_ALUNO));

CREATE TABLE MATERIAS (
ID_MATERIA INT NOT NULL IDENTITY(1,1),
NOME VARCHAR(45) NOT NULL,
COD_PROFESSOR INT,
COD_ALUNO INT,
CONSTRAINT PK_MATERIA PRIMARY KEY(ID_MATERIA),
CONSTRAINT FK_CODPROFESSOR FOREIGN KEY (ID_PROFESSOR) REFERENCES (PROFESSORES)
CONSTRAINT FK_CODALUNO FOREIGN KEY (ID_ALUNO) REFERENCES (ALUNOS));
